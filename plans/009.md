```typescript
export function camelCase(s: string): string {
  const words = s.split(/[\s\-_]+/).filter(Boolean)
  if (words.length === 0) return ''
  return words[0].toLowerCase() + words.slice(1).map(w => w[0].toUpperCase() + w.slice(1).toLowerCase()).join('')
}
```
