```typescript
export function kebabCase(s: string): string {
  return s
    .replace(/([a-z])([A-Z])/g, '$1-$2')
    .split(/[\s\-_]+/)
    .filter(Boolean)
    .map(w => w.toLowerCase())
    .join('-')
}
```
